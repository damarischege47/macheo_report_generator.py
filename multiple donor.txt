import os

# =========================
# CONFIGURATION
# =========================
API_KEY = os.getenv("MONDAY_API_KEY")
EMAIL_PASSWORD = os.getenv("MACHEO_EMAIL_PASSWORD")

if not API_KEY or not EMAIL_PASSWORD:
    raise EnvironmentError("Missing required environment variables.")

SMTP_SERVER = "smtp.gmail.com"
SMTP_PORT = 587

SENDER_EMAIL = "macheoreports@macheo.org"
RECIPIENT_EMAIL = "macheoreports@macheo.org"

CURRENT_YEAR = datetime.now().year

# =========================
# MONDAY API
# =========================
def monday_request(query: str) -> dict:
    response = requests.post(
        "https://api.monday.com/v2",
        json={"query": query},
        headers={"Authorization": API_KEY},
        timeout=30
    )
    response.raise_for_status()
    return response.json()


def get_board_items(board_id: str) -> list:
    query = f"""
    {{
      boards(ids: [{board_id}]) {{
        name
        items_page(limit: 500) {{
          items {{
            id
            name
            updated_at
          }}
        }}
      }}
    }}
    """
    data = monday_request(query)
    boards = data.get("data", {}).get("boards", [])
    if not boards:
        return []

    board = boards[0]
    items = board.get("items_page", {}).get("items", [])

    # filter only current year
    return [
        {**item, "board_name": board["name"]}
        for item in items
        if item.get("updated_at")
        and datetime.strptime(item["updated_at"], "%Y-%m-%dT%H:%M:%SZ").year == CURRENT_YEAR
    ]


# =========================
# DATA EXTRACTION
# =========================
def extract_summary(subitems: list) -> dict:
    success, challenges = [], []

    for sub in subitems:
        name = sub.get("name", "").lower()

        for asset in sub.get("assets", []):
            if not asset.get("name", "").endswith(".docx"):
                continue

            text = read_docx(asset["public_url"])
            if not text:
                continue

            if "success" in name:
                success.append(text)
            elif "challenge" in name:
                challenges.append(text)

    return {
        "Success": "\n\n".join(success) or "No recorded successes.",
        "Challenges": "\n\n".join(challenges) or "No recorded challenges."
    }


def read_docx(url: str) -> str:
    try:
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        doc = Document(io.BytesIO(r.content))
        return "\n".join(p.text.strip() for p in doc.paragraphs if p.text.strip())
    except Exception:
        return ""


# =========================
# REPORT GENERATION
# =========================
def generate_report(template: Document, interventions: list) -> io.BytesIO:
    buffer = io.BytesIO()
    template.save(buffer)
    buffer.seek(0)

    doc = Document(buffer)

    for i, item in enumerate(interventions, 1):
        doc.add_heading(f"{i}. {item['name']}", level=3)

        doc.add_heading("Success", level=4)
        doc.add_paragraph(item["summary"]["Success"])

        doc.add_heading("Challenges", level=4)
        doc.add_paragraph(item["summary"]["Challenges"])

    output = io.BytesIO()
    doc.save(output)
    output.seek(0)

    return output


# =========================
# EMAIL
# =========================
def send_email(file_stream: io.BytesIO, donor: str):
    msg = MIMEMultipart()
    msg["From"] = SENDER_EMAIL
    msg["To"] = RECIPIENT_EMAIL
    msg["Subject"] = f"Interventions Report – {donor}"

    body = f"""Dear Communication Team,

Please find attached the interventions report for {donor}.

Regards,  
Damaris
"""
    msg.attach(MIMEText(body, "plain"))

    part = MIMEApplication(file_stream.read(), Name=f"{donor}.docx")
    part["Content-Disposition"] = f'attachment; filename="{donor}.docx"'
    msg.attach(part)

    with smtplib.SMTP(SMTP_SERVER, SMTP_PORT) as server:
        server.starttls()
        server.login(SENDER_EMAIL, EMAIL_PASSWORD)
        server.send_message(msg)
